class TorchPipeline:
    def __init__(self, X_train, y_train, X_valid, y_valid, n_hidden, n_output, typs = 'binary', epochs = 500, lr = 0.01):
        assert typs in ['binary', 'multiclass']
        self.typs = typs
        
        epochs = epochs
        device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        model = self.ClassicNeuralNetwork(n_input = X_train.shape[1], n_hidden = n_hidden, n_output = n_output).to(device)
        
        criterion = None
        if self.typs == 'binary':
            criterion = nn.BCEWithLogitsLoss()
        else:
            criterion = nn.CrossEntropyLoss()
        
        optimizer = optimi.Adam(model.parameters(), lr = lr)

        X_train, X_valid = torch.tensor(X_train).to(device), torch.tensor(X_valid).to(device) 
        y_train, y_valid = torch.tensor(y_train).to(device), torch.tensor(y_valid).to(device)

        train_losses, train_accuracies, valid_losses, valid_accuracies = [], [], [], []

        for epoch in range(epochs):
            train_loss, train_accuracy, model = self.train_epoch(model, X_train, y_train, optimizer, criterion)
            valid_loss, valid_accuracy, = self.valid_epoch(model, X_valid, y_valid, criterion)

            if epoch % 50 == 0:
                print("Epoch: {}, Train Loss: {}, Valid Loss: {}, Train Accuracy {}, Valid Accuracy: {}".format(epoch, train_loss, valid_loss, train_accuracy, valid_accuracy))

            train_losses.append(train_loss)
            train_accuracies.append(train_accuracy)

            valid_losses.append(valid_loss)
            valid_accuracies.append(valid_accuracy)

        fig, axs = plt.subplots(2, 2, figsize = (15, 8))
        axs[0, 0].plot(train_losses, label = "Train Losses")
        axs[0, 0].set_title("Train Losses")
        
        axs[0, 1].plot(valid_losses, label = "Valid Losses")
        axs[0, 1].set_title("Valid Losses")
        
        axs[1, 0].plot(train_accuracies, label = "Train Accuracies")
        axs[1, 0].set_title("Train Accuracies")
        
        axs[1, 1].plot(valid_accuracies, label = "Valid Accuracies")
        axs[1, 1].set_title("Valid Accuracies")
        
        plt.show()
        
    class ClassicNeuralNetwork(nn.Module):
        def __init__(self, n_input, n_hidden, n_output):
            super().__init__()

            self.n_input  = n_input
            self.n_hidden = n_hidden
            self.n_output = n_output

            self.fc1 = nn.Linear(n_input, n_hidden, bias = False)
            self.fc2 = nn.Linear(n_hidden, n_output, bias = False)

            self.activation = nn.Tanh()

            self.sigmoid = nn.Sigmoid()
            self.softmax = nn.Softmax()

        def forward(self, x):

            x = self.activation(self.fc1(x))
            x = self.fc2(x)

            return x
        
    def train_epoch(self, model, X_train, y_train, optimizer, criterion):
        model.train()
        optimizer.zero_grad()

        output = model(X_train.float()).squeeze(1)
        
        preds = None
        if self.typs == 'binary':
            y_train = y_train.float()
            
            probas = torch.sigmoid(output)
            preds  = torch.round(probas)
        else:
            y_train = F.one_hot(y_train)
            y_train = torch.max(y_train, 1)[1]
            
            probas  = F.softmax(output)
            preds   = torch.round(probas)
            preds   = torch.argmax(preds, dim = 1)
            
        loss = criterion(output, y_train)

        loss.backward()
        optimizer.step()
            
        preds  = preds.cpu().detach().numpy()
        labels = y_train.cpu().detach().numpy()
        
        accuracy   = accuracy_score(labels, preds)
        train_loss = loss.item()

        return np.round(train_loss, 3), np.round(accuracy, 3), model
    
    def valid_epoch(self, model, X_valid, y_valid, criterion):
        model.eval()

        with torch.no_grad():
            output = model(X_valid.float()).squeeze(1)
            
            preds = None
            if self.typs == 'binary':
                y_valid = y_valid.float()
                probas = torch.sigmoid(output)
                preds  = torch.round(probas)
            else:
                y_valid = F.one_hot(y_valid)
                y_valid = torch.max(y_valid, 1)[1]
                
                probas  = F.softmax(output)
                preds   = torch.round(probas)
                preds   = torch.argmax(preds, dim = 1)
                
            loss   = criterion(output, y_valid)
                
            preds  = preds.cpu().detach().numpy()
            labels = y_valid.cpu().detach().numpy()
                
            accuracy   = accuracy_score(labels, preds)
            valid_loss = loss.item()

        return np.round(valid_loss, 3), np.round(accuracy, 3)

def display_posterior_weights(ppc):
    fig, axes = plt.subplots(2, 2, figsize = (17, 12))

    for neuron in range(ppc['w_1'].shape[1]):
        for layer in range(ppc['w_1'].shape[1]):
            posterior_weight = [ppc['w_1'][iteration][neuron][layer] for iteration in range(ppc['w_1'].shape[0])]

def BinaryPymcNetwork(input_data, output_data, n_hidden):
    init_1 = np.random.randn(input_data.shape[1], n_hidden).astype(float)
    init_2 = np.random.randn(n_hidden).astype(float)
    
    total_size = output_data.shape[0]
    
    with pm.Model() as nn:
        input_data  = pm.Data('input_data',  input_data)
        output_data = pm.Data('output_data', output_data)
        
        # Generating the weights based on prior init variables
        weights_1 = pm.Normal("w_1", mu = 0, sigma = 1, shape = (X.shape[1], n_hidden), testval = init_1)
        weights_2 = pm.Normal("w_2", mu = 0, sigma = 1, shape = (n_hidden,), testval = init_2)
    
        activation_1 = pm.math.tanh(pm.math.dot(input_data, weights_1))
        activation_2 = pm.math.sigmoid(pm.math.dot(activation_1, weights_2))
        
        output = pm.Bernoulli("out", activation_2, observed = output_data, total_size = total_size)
        
    return nn

Converting regular weights to random variables¶
    

  • A better way to think about this is attaching each weight between neurons a probability distribution
    • 

  • And in the end will sample values as outputs from all distributions
    • 

  • To remeber those are the priors
    • 














In [26]:



    


def model(x_data, y_data):
    log_softmax = nn.LogSoftmax(dim = 1)
    
    fc1_weight_prior = Normal(loc = torch.zeros_like(PyroModel.fc1.weight), scale = torch.ones_like(PyroModel.fc1.weight))
    fc2_weight_prior = Normal(loc = torch.zeros_like(PyroModel.fc2.weight), scale = torch.ones_like(PyroModel.fc2.weight))
    
    priors = {'fc1.weight': fc1_weight_prior, 'fc2.weight': fc2_weight_prior}
    
    lifted_module = pyro.random_module("module", PyroModel, priors)
    
    lifted_reg_model = lifted_module()
    
    lhat = log_softmax(lifted_reg_model(x_data))
    
    pyro.sample("obs", Categorical(logits = lhat), obs = y_data)



    















The guide function - Hardest thing to understand¶
    

  • The main idea here it's that the stand alone model function (after we attach random variables) knows only the prior distributions. 
    • 

  • To optimize those priors we need to update them based on training data, generating the posterior distributions that "do well" on the fitted data
    • 

  • After those posteriors are generated from optimizing the guide function, those will be used to sample models for predictions
    • 














In [27]:



    


def guide(x_data, y_data):
    softplus    = nn.Softplus()
    
    # First layer weight distribution priors
    fc1_weight_mu    = torch.randn_like(PyroModel.fc1.weight)
    fc1_weight_sigma = torch.randn_like(PyroModel.fc1.weight)
    
    fc1_weight_mu_param    = pyro.param("fc1_weight_mu", fc1_weight_mu)
    fc1_weight_sigma_param = softplus(pyro.param("fc1_weight_sigma", fc1_weight_sigma))
    
    fc1_weight_prior = Normal(loc = fc1_weight_mu_param, scale = fc1_weight_sigma_param)

    # Second layer weight distribution priors
    fc2_weight_mu    = torch.randn_like(PyroModel.fc2.weight)
    fc2_weight_sigma = torch.randn_like(PyroModel.fc2.weight)
    
    fc2_weight_mu_param    = pyro.param("fc2_weight_mu", fc2_weight_mu)
    fc2_weight_sigma_param = softplus(pyro.param("fc2_weight_sigma", fc2_weight_sigma))
    
    fc2_weight_prior = Normal(loc = fc2_weight_mu_param, scale = fc2_weight_sigma_param).independent(1)

    priors = {'fc1.weight': fc1_weight_prior, 'fc2.weight': fc2_weight_prior}
    
    lifted_module = pyro.random_module("module", PyroModel, priors)
    
    return lifted_module()

def CategoricalPymcNetwork(input_data, output_data, n_hidden, n_class):
    input_size = input_data.shape[1]
    total_size = output_data.shape[0]
    
    
    init_1 = np.random.randn(input_size, n_hidden).astype(float)
    init_2 = np.random.randn(n_hidden, n_hidden).astype(float)
    init_3 = np.random.randn(n_hidden, n_class).astype(float)
    

    with pm.Model() as nn:
        input_data  = pm.Data('input_data',  input_data)
        output_data = pm.Data('output_data', output_data)
        
        weights_1 = pm.Normal("w_1", mu = 0, sigma = 1, shape = (input_size, n_hidden), testval = init_1)
        weights_2 = pm.Normal("w_2", mu = 0, sigma = 1, shape = (n_hidden, n_hidden), testval = init_2)
        weights_3 = pm.Normal("w_3", mu = 0, sigma = 1, shape = (n_hidden, n_class), testval = init_3)
    
        activation_1 = pm.math.tanh(pm.math.dot(input_data, weights_1))
        activation_2 = pm.math.tanh(pm.math.dot(activation_1, weights_2))
        activation_3 = T.nnet.softmax(pm.math.dot(activation_2, weights_3))
        
        output = pm.Categorical("out", p = activation_3, observed = output_data, total_size = total_size)
        
    return nn
